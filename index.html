#!/usr/bin/env node // audit.js — sanity-check FIXTURES and KNOWN_RESULTS in index.html // Usage: node audit.js [path/to/index.html] // // Checks: // 1. Duplicate KNOWN_RESULTS entries (same team pair + date) // 2. Orphaned results — result has no matching fixture // 3. Wrong-date results — teams match a fixture but date doesn't // 4. Missing results — past fixtures with no result // 5. Stale live flags — live:true match kicked off > 3 h ago const fs = require('fs'); const path = require('path'); const htmlPath = process.argv[2] || path.join(__dirname, 'index.html'); const html = fs.readFileSync(htmlPath, 'utf8'); // ── Extract FIXTURES array ──────────────────────────────────────────────────── const fixturesMatch = html.match(/const FIXTURES\s*=\s*(\[[\s\S]*?\]);/); if (!fixturesMatch) { console.error('Could not find FIXTURES'); process.exit(1); } // ── Extract KNOWN_RESULTS array ─────────────────────────────────────────────── const resultsMatch = html.match(/const KNOWN_RESULTS\s*=\s*(\[[\s\S]*?\]);/); if (!resultsMatch) { console.error('Could not find KNOWN_RESULTS'); process.exit(1); } // Safe eval inside a sandbox — we only need plain object/array literals. function safeEval(src) { // Replace unquoted property shorthand like usa:true → handled by Function() return Function('"use strict"; return (' + src + ')')(); } const FIXTURES = safeEval(fixturesMatch[1]); const KNOWN_RESULTS = safeEval(resultsMatch[1]); // ── Normalisation (mirrors index.html) ─────────────────────────────────────── function normalizeName(n) { if (!n) return ''; return n.trim() .replace(/Türkiye|Turkey/i, 'Türkiye') .replace(/Côte d'Ivoire|Ivory Coast/i, 'Ivory Coast') .replace(/Bosnia[- &]+Herzegovina/i, 'Bosnia and Herzegovina') .replace(/DR Congo|Congo DR/i, 'Congo DR') .replace(/Czech Republic/i, 'Czechia') .replace(/Korea Republic|South Korea/i, 'South Korea') .replace(/Curaçao|Curacao/i, 'Curaçao') .replace(/United States|USA/i, 'USA'); } function scoreKey(t1, t2, date) { const [a, b] = [normalizeName(t1), normalizeName(t2)].sort(); return `${a} v ${b} ${date}`; } function fixtureKey(f) { return scoreKey(f.t1, f.t2, f.date); } // ── TODAY in PDT (UTC-7) ───────────────────────────────────────────────────── const nowPdt = new Date(Date.now() - 7 * 3600 * 1000); const todayPdt = nowPdt.toISOString().slice(0, 10); // ── Helpers ─────────────────────────────────────────────────────────────────── function parsePdtMins(pdt) { // e.g. "12:00 pm", "9:00 am" const m = pdt.match(/(\d+):(\d+)\s*(am|pm)/i); if (!m) return 0; let h = parseInt(m[1]); const min = parseInt(m[2]); const mer = m[3].toLowerCase(); if (mer === 'pm' && h !== 12) h += 12; if (mer === 'am' && h === 12) h = 0; return h * 60 + min; } // ── Build fixture lookup ────────────────────────────────────────────────────── const fixtureByKey = {}; // scoreKey → fixture const fixtureByPair = {}; // "NormT1|NormT2" → [fixture, ...] (order-independent) FIXTURES.forEach(f => { const key = fixtureKey(f); fixtureByKey[key] = f; const pair = [normalizeName(f.t1), normalizeName(f.t2)].sort().join('|'); (fixtureByPair[pair] = fixtureByPair[pair] || []).push(f); }); // ── RUN CHECKS ─────────────────────────────────────────────────────────────── let issues = 0; function warn(msg) { console.warn('⚠️ ', msg); issues++; } function info(msg) { console.log(' ', msg); } // ── 1. Duplicates in KNOWN_RESULTS ─────────────────────────────────────────── console.log('\n── 1. Duplicate KNOWN_RESULTS entries ──'); const seen = {}; // scoreKey → first index seen (may be 0, so check !== undefined) KNOWN_RESULTS.forEach((r, i) => { const key = scoreKey(r.t1, r.t2, r.date); if (seen[key] !== undefined) { warn(`Duplicate result: ${r.t1} vs ${r.t2} on ${r.date} (indices ${seen[key]} and ${i})`); } else { seen[key] = i; } }); if (issues === 0) console.log(' OK — no duplicates'); // ── 2 & 3. Orphaned / wrong-date results ───────────────────────────────────── console.log('\n── 2. Orphaned or wrong-date results ──'); let orphanIssues = 0; KNOWN_RESULTS.forEach((r, i) => { const key = scoreKey(r.t1, r.t2, r.date); const pair = [normalizeName(r.t1), normalizeName(r.t2)].sort().join('|'); if (fixtureByKey[key]) return; // perfect match — fine const candidates = fixtureByPair[pair] || []; if (candidates.length === 0) { warn(`[${i}] No fixture found for "${r.t1}" vs "${r.t2}" on any date`); } else { const dates = candidates.map(f => f.date).join(', '); warn(`[${i}] "${r.t1}" vs "${r.t2}" has result date ${r.date} but fixture date(s): ${dates}`); } orphanIssues++; }); if (orphanIssues === 0) console.log(' OK — all results match a fixture'); // ── 4. Past fixtures missing a result ──────────────────────────────────────── console.log('\n── 3. Past fixtures with no result ──'); let missingCount = 0; FIXTURES.forEach(f => { // Skip TBD fixtures (knockout placeholders) if (f.t1 === 'TBD' || f.t2 === 'TBD') return; // Skip future matches if (f.date > todayPdt) return; // If same date as today, only flag if kickoff was > 3 h ago if (f.date === todayPdt) { const kickoffMins = parsePdtMins(f.pdt || ''); const nowMins = nowPdt.getHours() * 60 + nowPdt.getMinutes(); if (nowMins - kickoffMins < 180) return; // too early to have a result } const key = fixtureKey(f); if (seen[key] === undefined) { warn(`No result for ${f.t1} vs ${f.t2} on ${f.date} (fixture id: ${f.id})`); missingCount++; } }); if (missingCount === 0) console.log(' OK — all past fixtures have results'); // ── 5. Stale live flags ─────────────────────────────────────────────────────── console.log('\n── 4. Stale live:true flags ──'); const STALE_HOURS = 3; let staleCount = 0; KNOWN_RESULTS.filter(r => r.live).forEach(r => { const fixture = fixtureByKey[scoreKey(r.t1, r.t2, r.date)]; if (!fixture) { warn(`live:true result for ${r.t1} vs ${r.t2} has no matching fixture`); staleCount++; return; } const kickoffMins = parsePdtMins(fixture.pdt || ''); const fixtureDateMs = new Date(fixture.date + 'T00:00:00Z').getTime() + kickoffMins * 60000 + 7 * 3600 * 1000; // PDT offset const hoursSinceKick = (Date.now() - fixtureDateMs) / 3_600_000; if (hoursSinceKick > STALE_HOURS) { warn(`STALE LIVE FLAG: ${r.t1} vs ${r.t2} (${r.date}) kicked off ${hoursSinceKick.toFixed(1)}h ago`); staleCount++; } }); if (staleCount === 0) console.log(' OK — no stale live flags'); // ── Summary ─────────────────────────────────────────────────────────────────── console.log(`\n${'─'.repeat(50)}`); if (issues === 0) { console.log(`✅ All checks passed — ${FIXTURES.length} fixtures, ${KNOWN_RESULTS.length} results`); } else { console.log(`❌ ${issues} issue(s) found — see warnings above`); } console.log();
